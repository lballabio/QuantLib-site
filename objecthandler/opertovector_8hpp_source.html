81  template <class T>
82  std::vector<T> operToVectorImpl(
83  const ConvertOper &xVector,
84  const std::string &paramName) {
85 
86  try {
87 
88  if (xVector.missing()) return std::vector<T>();
89 
90  OH_REQUIRE(!xVector.error(), "input value has type=error");
91 
92  const OPER *xMulti;
93  Xloper xCoerce; // Freed automatically
94  X xSplit; // Freed automatically
95 
96  if (xVector->xltype == xltypeMulti) {
97  xMulti = xVector.get();
98  } else if (xVector->xltype == xltypeStr) {
99  splitOper(xVector.get(), &xSplit.o);
100  xMulti = &xSplit.o;
101  } else {
102  Excel(xlCoerce, &xCoerce, 2, xVector.get(), TempInt(xltypeMulti));
103  xMulti = &xCoerce;
104  }
105 
106  std::vector<T> ret;
107  ret.reserve(xMulti->val.array.rows * xMulti->val.array.columns);
108  for (int i=0; i<xMulti->val.array.rows * xMulti->val.array.columns; ++i) {
109  ret.push_back(convert2<T>(ConvertOper(xMulti->val.array.lparray[i])));
110  }
111 
112  return ret;
113  } catch (const std::exception &e) {
114  OH_FAIL("operToVectorImpl: error converting parameter '" << paramName
115  << "' to type '" << typeid(T).name() << "' : " << e.what());
116  }
117  }
