46  template <class T>
47  std::vector<std::vector<T> > operToMatrixImpl(
48  const ConvertOper &xMatrix,
49  const std::string &paramName) {
50 
51  try {
52  if (xMatrix.missing()) return std::vector<std::vector<T> >();
53 
54  OH_REQUIRE(!xMatrix.error(), "input value has type=error");
55 
56  const OPER *xMulti;
57  Xloper xCoerce; // Freed automatically
58 
59  if (xMatrix->xltype == xltypeMulti)
60  xMulti = xMatrix.get();
61  else {
62  Excel(xlCoerce, &xCoerce, 2, xMatrix.get(), TempInt(xltypeMulti));
63  xMulti = &xCoerce;
64  }
65 
66  std::vector<std::vector<T> > ret;
67  ret.reserve(xMulti->val.array.rows);
68  for (int i=0; i<xMulti->val.array.rows; ++i) {
69  std::vector<T> row;
70  row.reserve(xMulti->val.array.columns);
71  for (int j=0; j<xMulti->val.array.columns; ++j) {
72  row.push_back(convert2<T>(ConvertOper(xMulti->val.array.lparray[i * xMulti->val.array.columns + j])));
73  }
74  ret.push_back(row);
75  }
76 
77  return ret;
78  } catch (const std::exception &e) {
79  OH_FAIL("operToMatrixImpl: error converting parameter '" << paramName
80  << "' to type '" << typeid(T).name() << "' : " << e.what());
81  }
82  }

86  std::vector<std::vector<T> > fpToMatrix(const FP &fpMatrix) {
87  std::vector<std::vector<T> > ret;
88  ret.reserve(fpMatrix.rows);
89  for (int i=0; i<fpMatrix.rows; ++i) {
90  std::vector<T> row;
91  row.reserve(fpMatrix.columns);
92  for (int j=0; j<fpMatrix.columns; ++j)
93  row.push_back(fpMatrix.array[i * fpMatrix.columns + j]);
94  ret.push_back(row);
95  }
