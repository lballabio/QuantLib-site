65  template<class LoopFunction, class InputType, class OutputType>
66  void loop(
67  const boost::shared_ptr<FunctionCall> &functionCall,
68  LoopFunction &loopFunction,
69  OPER *xIn,
70  XLOPER &xOut) {
71 
72  // FIXME - xTemp may not be cleaned up properly in the event of an exception.
73  OPER xTemp, *xMulti;
74  bool excelToFree = false;
75  bool xllToFree = false;
76 
77  // If the input is an array then take its address & carry on
78  if (xIn->xltype == xltypeMulti) {
79  xMulti = xIn;
80  // If the input is a list then call split on it
81  } else if (isList(xIn)) {
82  splitOper(xIn, &xTemp);
83  xMulti = &xTemp;
84  xllToFree = true;
85  // If the input is a scalar then just call the function once & return
86  } else if (xIn->xltype == xltypeNum
87  || xIn->xltype == xltypeBool
88  || xIn->xltype == xltypeStr) {
89  LoopIteration<LoopFunction, InputType, OutputType>()(
90  loopFunction, *xIn, xOut, true);
91  return;
92  // Some other input (e.g. a reference) - try to convert to an array
93  } else {
94  Excel(xlCoerce, &xTemp, 2, xIn, TempInt(xltypeMulti));
95  xMulti = &xTemp;
96  excelToFree = true;
97  }
98 
99  xOut.val.array.rows = xMulti->val.array.rows;
100  xOut.val.array.columns = xMulti->val.array.columns;
101  int numCells = xMulti->val.array.rows * xMulti->val.array.columns;
102  xOut.val.array.lparray = new XLOPER[numCells];
103  xOut.xltype = xltypeMulti | xlbitDLLFree;
104 
105  int errorCount = 0;
106  std::ostringstream err;
107  LoopIteration<LoopFunction, InputType, OutputType> loopIteration;
108  for (int i=0; i<numCells; ++i) {
109  try {
110  loopIteration(loopFunction,
111  xMulti->val.array.lparray[i],
112  xOut.val.array.lparray[i],
113  false);
114  } catch (const std::exception &e) {
115  xOut.val.array.lparray[i].xltype = xltypeErr;
116  xOut.val.array.lparray[i].val.err = xlerrNum;
117 
118  if (errorCount > ERROR_LIMIT) {
119  // Limit exceeded. Take no action. For performance reasons we test
120  // this case first since it's most common on big loop w/many errors
121  ;
122  } else if (errorCount < ERROR_LIMIT) {
123  err << std::endl << std::endl
124  << "iteration #" << i << " - " << e.what();
125  errorCount++;
126  } else { // errorCount == ERROR_LIMIT
127  err << std::endl << std::endl
128  << "iteration #" << i << " - " << e.what()
129  << std::endl << std::endl
130  << "Count of failed iterations in looping function hit "
131  << "limit of " << ERROR_LIMIT + 1 << " - logging discontinued";
132  errorCount++;
133  }
134  }
135  }
136 
137  if (errorCount)
138  RepositoryXL::instance().logError(err.str(), functionCall);
139 
140  // Free memory
141  if (excelToFree) {
142  Excel(xlFree, 0, 1, &xTemp);
143  } else if (xllToFree) {
144  freeOper(&xTemp);
145  }
146 
147  }
148 
149 }
