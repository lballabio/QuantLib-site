118  inline void Observable::notifyObservers() {
119  bool successful = true;
120  std::string errMsg;
121  for (iterator i=observers_.begin(); i!=observers_.end(); ++i) {
122  try {
123  (*i)->update();
124  } catch (std::exception& e) {
125  // quite a dilemma. If we don't catch the exception,
126  // other observers will not receive the notification
127  // and might be left in an incorrect state. If we do
128  // catch it and continue the loop (as we do here) we
129  // lose the exception. The least evil might be to try
130  // and notify all observers, while raising an
131  // exception if something bad happened.
132  successful = false;
133  errMsg = e.what();
134  } catch (...) {
135  successful = false;
136  }
137  }
138  OH_REQUIRE(successful,
139  "could not notify one or more observers: " << errMsg);
140  }
