93  template <typename KeyClass>
94  inline void Registry<KeyClass>::registerType(
95  const std::string &mapID,
96  const KeyClass &typeID,
97  void *type) const {
98  AllTypeMap &allTypeMap = getAllTypesMap();
99  TypeMapPtr typeMapPtr;
100  typename AllTypeMap::const_iterator i = allTypeMap.find(mapID);
101  if (i == allTypeMap.end()) {
102  typeMapPtr = TypeMapPtr(new TypeMap);
103  allTypeMap[mapID] = typeMapPtr;
104  } else {
105  typeMapPtr = i->second;
106  }
107  (*typeMapPtr)[typeID] = type;
108  }
