145  inline void Object::dump(std::ostream &out) {
146  out << std::endl;
147  std::set<std::string> propertyNames = this->propertyNames();
148  for (std::set<std::string>::const_iterator i = propertyNames.begin();
149  i != propertyNames.end(); ++i) {
150  std::string propertyName = *i;
151  property_t propertyValue = this->propertyValue(propertyName);
152  out << "property = " << std::left << std::setw(logColumnWidth) << propertyName;
153  out << " value = " << std::left << std::setw(logColumnWidth) << propertyValue << std::endl;
154  }
155  out << "Permanent = " << std::left << std::setw(logColumnWidth)
156  << std::boolalpha << permanent_ << std::endl;
157  out << std::endl;
158  }
